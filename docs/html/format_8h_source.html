177  {
178  const auto endIndex = format.find('}');
179  context.assertTrue(endIndex != bzd::StringView::npos, "Missing closing '}' for the replacement field");
180  auto metadataStr = format.subStr(0, endIndex + 1);

182  Metadata metadata;
183 
184  // Look for the index
185  metadata.index = parseIndex(context, metadataStr, current);
186  context.assertTrue(metadataStr.size() > 0, "Replacement field format ended abruptly (after parseIndex)");
187 
188  // Parse sign: [sign]
189  parseSign(context, metadataStr, metadata);
190  context.assertTrue(metadataStr.size() > 0, "Replacement field format ended abruptly (after parseSign)");
191 
192  // Parse alternate form [#]
193  if (metadataStr.front() == '#')
194  {
195  metadata.alternate = true;
196  metadataStr.removePrefix(1);
197  context.assertTrue(metadataStr.size() > 0, "Replacement field format ended abruptly '#')");
198  }
199

201  if (metadataStr.front() == '.')
202  {
203  metadataStr.removePrefix(1);
204  metadata.isPrecision = parseUnsignedInteger(metadataStr, metadata.precision);

206  }
207 
208  // Parse type [type]

210  {
211  switch (metadataStr.front())
212  {
213  case 'b':
214  metadata.format = Metadata::Format::BINARY;
215  break;
216  case 'd':
217  metadata.format = Metadata::Format::DECIMAL;
218  break;
219  case 'o':
220  metadata.format = Metadata::Format::OCTAL;
221  break;
222  case 'x':

224  break;
225  case 'X':

227  break;
228  case 'f':

230  break;

236  break;
